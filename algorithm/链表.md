# 链表

## 带头结点双向循环链表

```cpp
struct node
    {
        int key, val;
        node* pre, * next;
        node(int x, int y) :key(x), val(y) {
            pre = nullptr;
            next = nullptr;
        }
    };
    struct list
    {
        node* root;
        int size;
        list()
        {
            size = 0;
            root = new node(-1, -1);
        }
        node* begin()
        {
            return root->next;
        }
        node* end()
        {
            return root->pre;
        }
        node* push_back(int x, int y)
        {
            size++;
            node* tmp = new node(x, y);
            tmp->next = root;
            tmp->pre = root->pre;
            tmp->next->pre = tmp;
            if (tmp->pre != nullptr)tmp->pre->next = tmp;
            if (root->next == nullptr)
            {
                root->next = tmp;
                tmp->pre = root;
            }
            return tmp;
        }
        node* push_front(int x, int y)
        {
            size++;
            node* tmp = new node(x, y);
            tmp->next = root->next;
            tmp->pre = root;
            tmp->pre->next = tmp;
            if (tmp->next != nullptr)tmp->next->pre = tmp;
            if (root->pre == nullptr)
            {
                root->pre = tmp;
                tmp->next = root;
            }
            return tmp;
        }
        void pop_back()
        {
            erase(root->pre);
        }
        void erase(node* p)
        {
            size--;
            if (p->pre != nullptr)p->pre->next = p->next;
            if (p->next != nullptr)p->next->pre = p->pre;
            delete p;
        }
    };
```
